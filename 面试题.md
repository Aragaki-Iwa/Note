1.一只青蛙一次可以跳上1级台阶，也可以跳上2级。求该青蛙跳上一个n级的台阶总共有多少种跳法（先后次序不同算不同的结果）
实际就是斐波那契数列，f[n] = f[n-1]+f[n-2] n>2.递归的话会有太多重复计算，速度慢，用迭代
    
int jumpFloor(int n)
{
    if(n==0||n==1||n==2)
    {
      return n;
    }
    int sum = 0;
    int t1 = 1;
    int t2 = 2;
    while(n>2)
    {
      sum = t1 + t2;
      t1 = t2;
      t2 = sum;
      --n;
    }
    return sum;
}

2.1 从上往下打印出二叉树的每个节点，同层节点从左至右打印

vector<int> PrintFromTopToBottom(TreeNode* root)
{
    //用队列
    vector<int> nodes;
    queue<TreeNode*> que;
    
    que.push(root);
    while(!que.empty())
    {
        root = que.front();que.pop();
        if(root==NULL)
            continue;
        nodes.push_back(root->val);
        que.push(root->left);
        que.push(root->right);
    }
    return nodes;
}
